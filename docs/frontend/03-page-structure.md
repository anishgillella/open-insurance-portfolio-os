## Layout System

### Root Layout

```tsx
// app/layout.tsx
import { Inter, JetBrains_Mono } from 'next/font/google';
import { Providers } from '@/components/Providers';
import { AppShell } from '@/components/layouts/AppShell';
import '@/styles/globals.css';

const inter = Inter({
  subsets: ['latin'],
  variable: '--font-sans',
});

const jetbrains = JetBrains_Mono({
  subsets: ['latin'],
  variable: '--font-mono',
});

export const metadata = {
  title: 'Open Insurance',
  description: 'AI-powered insurance portfolio management',
};

export default function RootLayout({ children }) {
  return (
    <html lang="en" className={`${inter.variable} ${jetbrains.variable}`}>
      <body>
        <Providers>
          <AppShell>{children}</AppShell>
        </Providers>
      </body>
    </html>
  );
}
```

### Property Sub-Layout

```tsx
// app/properties/[id]/layout.tsx
import { PropertyHeader } from '@/components/features/properties/PropertyHeader';
import { PropertyTabs } from '@/components/features/properties/PropertyTabs';

export default function PropertyLayout({ children, params }) {
  return (
    <div>
      <PropertyHeader propertyId={params.id} />
      <PropertyTabs propertyId={params.id} />
      <div className="mt-6">{children}</div>
    </div>
  );
}
```

```typescript
const breakpoints = {
  sm: '640px',   // Mobile landscape
  md: '768px',   // Tablet
```

### Mobile Adaptations

- Sidebar collapses to bottom navigation
- Property cards stack vertically
- 3D visualizations simplified or replaced with 2D
- Tables become card lists
- Chat becomes full-screen
